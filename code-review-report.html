type RateLimitConfig = {
  maxEvents: number;
  windowMs: number;
};

function createRateLimiter(config: RateLimitConfig) {
  return {
    attempts: new Map(),

    isAllowed(socketId: string): boolean {
      const now = Date.now();
      const key = socketId;

      if (!this.attempts.has(key)) {
        this.attempts.set(key, [now]);
        return true;
      }

      const timestamps = this.attempts.get(key)!;
      const recentAttempts = timestamps.filter(
        (t) => now - t < config.windowMs
      );

      if (recentAttempts.length >= config.maxEvents) {
        return false;
      }

      recentAttempts.push(now);
      this.attempts.set(key, recentAttempts);
      return true;
    }
  };
}

  io: Server,
  roomId: string,
  onTimerExpired: (io: Server, roomId: string) => void,

  // Always clear old timer first
  const existingInterval = timerIntervals.get(roomId);
  if (existingInterval) {
    clearInterval(existingInterval);
    timerIntervals.delete(roomId);
  }

  const interval = setInterval(() => {
    try {
      const state = getGameState(roomId);
      
      // Stop timer if state is gone
      if (!state) {
        clearInterval(interval);
        timerIntervals.delete(roomId);
        return;

        io.to(roomId).emit("game:timer", {
          seconds: remaining,
          startTime: state.startTime,
        });
      }

      // Clean up when timer expires
      if (remaining <= 0) {
        clearInterval(interval);
        timerIntervals.delete(roomId);
        onTimerExpired(io, roomId);
      }
    } catch (error) {
      console.error(`Timer error for ${roomId}:`, error);
      clearInterval(interval);
      timerIntervals.delete(roomId);

      timerIntervals.delete(roomId);

}

class RoomCache {
  private cache = new Map();
  private dirty = new Set();

  async get(roomId: string): Promise {
    // ✅ Check memory cache first
    if (this.cache.has(roomId)) {
      return this.cache.get(roomId)!;
    }

    // Fetch from Redis if not cached
    const raw = await redis.get(
      KEYS.room(roomId)
    );
    if (!raw) return null;

    // Parse once and cache
    const room = JSON.parse(raw);
    this.cache.set(roomId, room);
    return room;
  }

  set(room: Room): void {
    // Keep in-memory version
    this.cache.set(room.id, room);
    // Mark for batch save
    this.dirty.add(room.id);
  }

  async flush(): Promise {
    // ✅ Batch save all dirty rooms at once
    const updates: Array<[string, string]> = [];
    
    for (const roomId of this.dirty) {
      const room = this.cache.get(roomId);
      if (room) {
        updates.push([
          KEYS.room(roomId),
          JSON.stringify(room),
        ]);
      }
    }

    // Single Redis call for all updates
    if (updates.length > 0) {
      await redis.mset(updates.flat());
    }

    this.dirty.clear();
  }
}

const roomCache = new RoomCache();

export function useGameSocket() {
  const [gameState, setGameState] = useState(null);
  const [timer, setTimer] = useState(0);
  const [phase, setPhase] = useState(null);

  useEffect(() => {
    // ✅ Each handler is small and focused
    const handlers = {
      onGameStarted: () => {
        setGameState({ ...EMPTY_GAME_STATE });
      },
      onRoundStart: (data: RoundStartPayload) => {
        setGameState((prev) =>
          prev ? { ...prev, currentCards: data.cards } : null
        );
      },
      onPhaseChanged: (data: PhaseChangedPayload) => {
        setPhase(data.phase);
        setGameState((prev) =>
          prev ? { ...prev, phase: data.phase } : null
        );
      },
      onTimer: (data: { seconds: number }) => {
        setTimer(data.seconds);
      },
    };

    // Attach listeners
    socket.on("game:started", handlers.onGameStarted);
    socket.on("game:round-start", handlers.onRoundStart);
    socket.on("game:phase-changed", handlers.onPhaseChanged);
    socket.on("game:timer", handlers.onTimer);

    // Cleanup
    return () => {
      socket.off("game:started", handlers.onGameStarted);
      socket.off("game:round-start", handlers.onRoundStart);
      socket.off("game:phase-changed", handlers.onPhaseChanged);
      socket.off("game:timer", handlers.onTimer);
    };
  }, []);

  return { gameState, timer, phase };
}

export async function joinRoom(
  roomId: string,
  player: Player,
  code?: string
): Promise {
  // Read is still individual
  const raw = await redis.get(KEYS.room(roomId));
  if (!raw) {
    return { success: false, error: "Not found" };
  }

  const room: Room = JSON.parse(raw);

  // Validate...
  if (room.players.length >= room.maxPlayers) {
    return { success: false, error: "Room full" };
  }

  room.players.push(player);

  // ✅ Batch writes in one pipeline
  const pipeline = redis.pipeline();
  
  pipeline.set(
    KEYS.room(room.id),
    JSON.stringify(room)
  );
  
  pipeline.set(
    KEYS.playerRoom(player.id),
    room.id
  );
  
  // ✅ Execute all at once
  await pipeline.exec();

  return { success: true, room };
}

// Result: 1 Redis round trip instead of 2!
// Less latency, better throughput
